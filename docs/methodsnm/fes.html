  8class FESpace(ABC):
  9    """
 10    Abstract base class for finite element spaces.
 11    """
 12    ndof = None
 13    mesh = None
 14    def __init__(self, mesh):
 15        """Initialize base FESpace.
 16
 17        Note: the base class does not set up DOFs or finite elements.
 18        Subclasses should override and set attributes like ``ndof``,
 19        ``mesh`` and element templates (e.g. ``fe``, ``sfe``) as needed.
 20
 21        Parameters
 22        ----------
 23        mesh : object
 24            Mesh-like object providing geometry, connectivity and
 25            boundary information used by subclasses.
 26        """
 27        # intentionally minimal: subclasses set specific attributes
 28        self.mesh = mesh
 29
 30    @abstractmethod
 31    def _finite_element(self, elnr):
 32        """Return the finite element object used on the given element.
 33
 34        This is an abstract method that must be implemented by subclasses.
 35
 36        Parameters
 37        ----------
 38        elnr : int
 39            Local element index (0-based) within the mesh.
 40
 41        Returns
 42        -------
 43        object
 44            A finite-element descriptor (object implementing shape
 45            function evaluation used elsewhere in the codebase).
 46        """
 47        raise Exception("Not implemented - Base class should not be used")
 48
 49    @abstractmethod
 50    def _element_dofs(self, elnr):
 51        """Return the global DOF indices associated with an element.
 52
 53        Subclasses must provide an array-like of integer DOF indices
 54        that describe the mapping from local element DOFs to global
 55        system-wide DOF numbering.
 56
 57        Parameters
 58        ----------
 59        elnr : int
 60            Local element index.
 61
 62        Returns
 63        -------
 64        list or ndarray
 65            Integer indices of global DOFs for the element.
 66        """
 67        raise Exception("Not implemented - Base class should not be used")
 68
 69    def _bndry_finite_element(self, elnr):
 70        """Return the 1D/face finite element used on a boundary element.
 71
 72        Default implementation raises when boundary evaluation is
 73        unsupported. Subclasses that support boundary evaluation should
 74        override this method.
 75
 76        Parameters
 77        ----------
 78        elnr : int
 79            Local boundary element index.
 80        """
 81        raise Exception("_bndry_finite_element not implemented - Base class should not be used")
 82
 83    def _bndry_element_dofs(self, elnr):
 84        """Return global DOF indices for a boundary element.
 85
 86        This method should mirror ``_element_dofs`` but for boundary
 87        elements (edges/faces on the mesh boundary). If boundary
 88        evaluation is not supported the default implementation raises.
 89
 90        Parameters
 91        ----------
 92        elnr : int
 93            Local boundary element index.
 94        """
 95        raise Exception("_bndry_element_dofs not implemented - Base class should not be used")
 96
 97    def finite_element(self, elnr, bndry=False):
 98        """
 99        Returns the finite element for the given element number.
100        """
101        if bndry:
102            return self._bndry_finite_element(elnr)
103        else:
104            return self._finite_element(elnr)
105
106    def element_dofs(self, elnr, bndry=False):
107        """
108        Returns the dofs for the given element number.
109        """
110        if bndry:
111            return self._bndry_element_dofs(elnr)
112        else:
113            return self._element_dofs(elnr)
114        
115    def boundary_vertices(self):
116        """Return the set of vertex indices that lie on the boundary.
117
118        Returns
119        -------
120        set
121            Set of integer vertex indices belonging to the mesh boundary.
122        """
123        return set(self.mesh.bndry_vertices)
124
125    def initial_vertices(self):
126        """Return the set of vertices on the initial/bottom boundary.
127
128        For time-space meshes this represents the 'initial time'
129        boundary. If the mesh does not provide this attribute an
130        AttributeError will be raised by the caller.
131        """
132        return set(self.mesh.initial_bndry_vertices)
133
134    def top_vertices(self):
135        """Return the set of vertices on the top boundary (final time).
136
137        Useful for applying time-dependent boundary conditions or for
138        excluding top vertices from certain boundary constraint sets.
139        """
140        return set(self.mesh.top_bndry_vertices)
141    
142    def boundary_dofs_excluding_top(self):
143        """Return boundary DOFs excluding those on the top boundary.
144
145        This is a convenience for problems where the top boundary should
146        remain free (e.g. time-marching where final-time conditions are
147        handled separately).
148
149        Returns
150        -------
151        list
152            Sorted list of boundary DOF indices not on the top boundary.
153        """
154        v = self.boundary_vertices() - self.top_vertices()
155        return sorted(self.boundary_dofs_from_vertex_set(v))
156
157    def boundary_dofs_from_vertex_set(self, vset):
158        """Map a set of vertex indices to boundary DOF indices.
159
160        Default behavior assumes one DOF per vertex and returns a
161        sorted list of the provided vertex indices. Subclasses that
162        have additional DOFs (edge/face DOFs) may override this method
163        to include those DOFs as well.
164
165        Parameters
166        ----------
167        vset : iterable
168            Iterable of integer vertex indices.
169
170        Returns
171        -------
172        list
173            Sorted list of integer DOF indices corresponding to the
174            provided vertices.
175        """
176        return sorted(vset)

206
207    def __init__(self, mesh, periodic=False):
208        super().__init__(mesh)
209        self.periodic = periodic
210        if periodic:

212        else:

220

222        element, so ``elnr`` is unused.

225
226    def _element_dofs(self, elnr):
227        """Return global DOF indices for a segment element.
228

230        indices making up the edge. For periodic meshes the last edge
231        shares the first global DOF (index 0).
232        """
233        dofs = self.mesh.edges[elnr]

240        """Return DOFs for a boundary element on a 1D segment space.
241
242        For periodic spaces the boundary is identified with the single
243        shared DOF (index 0). Otherwise return the boundary element
244        vertex indices as provided by the mesh.
245        """
246        dofs = self.mesh.elements(bndry=True)[bndry_elnr]
247        if self.periodic:
248            return [0]
249        else:
250            return dofs

306        offset = self.nv + elnr*(self.order-1)

407class P3_Triangle_Space(FESpace):
408
409    def __init__(self, mesh):
410        self.ndof = len(mesh.points) + 2 * len(mesh.edges) + len(mesh.faces)
411        self.nv = len(mesh.points)
412        self.ned = len(mesh.edges)
413        self.nf = len(mesh.faces)
414        self.mesh = mesh
415        self.fe = P3_Triangle_FE()
416        self.sfe = Lagrange_Segment_FE(order=3)
417
418    def _finite_element(self, elnr):
419        """Return the cubic triangle FE template (P3)."""
420        return self.fe
421
422    def _element_dofs(self, elnr):
423        """Return global DOFs for a P3 triangle element.
424
425        Local DOF ordering (length 10):
426        - indices 0..2 : vertex DOFs
427        - indices 3..8 : two DOFs per edge (orientation-sensitive)
428        - index 9      : cell interior DOF
429        """
430        dofs = np.empty(10, dtype=int)
431        vnums = self.mesh.elements()[elnr]
432        dofs[0:3] = vnums
433        j = 3
434        enums = self.mesh.faces2edges[elnr]
435        ref_verts_list = [[1,2],[0,2],[0,1]]
436        for ref_edge, edge in enumerate(enums):
437            tverts = vnums[ref_verts_list[ref_edge]]
438            if (tverts[0] < tverts[1]):
439                dofs[j] = self.nv + 2*edge + 0
440                j += 1
441                dofs[j] = self.nv + 2*edge + 1
442                j += 1
443            else:
444                dofs[j] = self.nv + 2*edge + 1
445                j += 1
446                dofs[j] = self.nv + 2*edge + 0
447                j += 1
448        dofs[9] = self.nv + 2 * self.ned + elnr
449        return dofs
450
451    def _bndry_finite_element(self, bndry_elnr):
452        """Return the cubic segment FE used on triangle boundaries."""
453        return self.sfe
454
455    def _bndry_element_dofs(self, bndry_elnr):
456        """Return DOFs for a boundary edge in a P3 triangle mesh.
457
458        The ordering is: left vertex, two ordered edge DOFs, right vertex.
459        """
460        verts = self.mesh.elements(bndry=True)[bndry_elnr]
461        edge = self.mesh.bndry_edges[bndry_elnr]
462        return [verts[0], self.nv + 2 * edge, self.nv + 2 * edge + 1, verts[1]]

527class P2_Hypertriangle_Space(FESpace):
528    """
529    P2 space on 4D simplex mesh.
530    DOFs: 1 per vertex + 1 per edge of each hyper-tetrahedron.
531    """
532
533    def __init__(self, mesh):
534        self.mesh = mesh
535        self._build_edges_from_hypercells()
536        self.nv = len(mesh.points)
537        self.ne = len(self.mesh.edges)
538        self.ndof = self.nv + self.ne
539        self.fe = P2_Hypertriangle_FE()
540
541    def _build_edges_from_hypercells(self):
542        """
543        Build global edge list and hypercells->edges connectivity
544        from hypercells (each with 5 vertices).
545        """
546        # if edges already exist, reuse them
547        if getattr(self.mesh, "edges", None) is not None and \
548           getattr(self.mesh, "hypercells2edges", None) is not None:
549            return
550
551        cells = self.mesh.elements()   # hypercells, shape (ncells, 5)
552        edge_dict = {}                 # (v1,v2) -> edge index
553        cells2edges = []
554
555        for el in cells:
556            verts = list(el)
557            local_edges = []
558            # all pairs of 5 vertices -> 10 edges
559            for a in range(5):
560                for b in range(a+1, 5):
561                    v1, v2 = verts[a], verts[b]
562                    key = (v1, v2) if v1 < v2 else (v2, v1)
563                    if key not in edge_dict:
564                        edge_dict[key] = len(edge_dict)
565                    local_edges.append(edge_dict[key])
566            cells2edges.append(local_edges)
567
568        # store on mesh for reuse
569        self.mesh.edges = np.array(list(edge_dict.keys()), dtype=int)
570        self.mesh.hypercells2edges = np.array(cells2edges, dtype=int)
571
572    def _finite_element(self, elnr):
573        """Return the P2 hypertriangle finite element template."""
574        return self.fe
575
576    def _element_dofs(self, elnr):
577        """
578        Local DOF numbering:
579        - first 5: vertex dofs (like P1)
580        - last 10: edge dofs (global index shifted by nv)
581        """
582        verts = self.mesh.elements()[elnr]
583        edge_ids = self.mesh.hypercells2edges[elnr]
584        return np.concatenate([verts, self.nv + edge_ids])
585    
586    def boundary_dofs(self):
587        """Return all boundary DOF indices for the P2 hypertriangle space.
588
589        The boundary DOFs consist of vertex DOFs lying on the boundary
590        plus any edge DOFs whose both vertices belong to the boundary
591        vertex set.
592
593        Returns
594        -------
595        list
596            Sorted list of integer DOF indices on the boundary.
597        """
598        v = self.boundary_vertices()
599        return sorted(
600            list(v) + self._edge_boundary_dofs(v)
601        )
602    
603    def boundary_dofs_from_vertex_set(self, vset):
604        """Return boundary DOFs (vertices + edges) restricted to vset.
605
606        Parameters
607        ----------
608        vset : iterable
609            Iterable/collection of vertex indices considered to be on a
610            boundary subset.
611
612        Returns
613        -------
614        list
615            Sorted list of corresponding vertex and edge DOF indices.
616        """
617        v = sorted(vset)
618        e = self._edge_boundary_dofs(vset)
619        return sorted(v + e)
620
621    def _edge_boundary_dofs(self, vertex_set):
622        """Helper: return edge DOFs for edges with both vertices in set.
623
624        Edge DOFs are stored after the vertex DOFs and thus are
625        returned as ``nv + edge_index``.
626        """
627        nv = self.nv
628        edge_dofs = []
629        for eid, (v1, v2) in enumerate(self.mesh.edges):
630            if v1 in vertex_set and v2 in vertex_set:
631                edge_dofs.append(nv + eid)
632        return edge_dofs

410    def __init__(self, mesh):
411        self.ndof = len(mesh.points) + 2 * len(mesh.edges) + len(mesh.faces)
412        self.nv = len(mesh.points)
413        self.ned = len(mesh.edges)

419    def _finite_element(self, elnr):

421        return self.fe

423    def _element_dofs(self, elnr):
424        """Return global DOFs for a P3 triangle element.
425
426        Local DOF ordering (length 10):
427        - indices 0..2 : vertex DOFs
428        - indices 3..8 : two DOFs per edge (orientation-sensitive)
429        - index 9      : cell interior DOF
430        """
431        dofs = np.empty(10, dtype=int)
432        vnums = self.mesh.elements()[elnr]
433        dofs[0:3] = vnums
434        j = 3
435        enums = self.mesh.faces2edges[elnr]
436        ref_verts_list = [[1,2],[0,2],[0,1]]
437        for ref_edge, edge in enumerate(enums):
438            tverts = vnums[ref_verts_list[ref_edge]]
439            if (tverts[0] < tverts[1]):
440                dofs[j] = self.nv + 2*edge + 0
441                j += 1
442                dofs[j] = self.nv + 2*edge + 1
443                j += 1
444            else:
445                dofs[j] = self.nv + 2*edge + 1
446                j += 1

538        self.ne = len(self.mesh.edges)

542    def _build_edges_from_hypercells(self):
543        """
544        Build global edge list and hypercells->edges connectivity
545        from hypercells (each with 5 vertices).
546        """
547        # if edges already exist, reuse them
548        if getattr(self.mesh, "edges", None) is not None and \
549           getattr(self.mesh, "hypercells2edges", None) is not None:
550            return
551
552        cells = self.mesh.elements()   # hypercells, shape (ncells, 5)
553        edge_dict = {}                 # (v1,v2) -> edge index
554        cells2edges = []
555
556        for el in cells:
557            verts = list(el)
558            local_edges = []
559            # all pairs of 5 vertices -> 10 edges
560            for a in range(5):
561                for b in range(a+1, 5):
562                    v1, v2 = verts[a], verts[b]
563                    key = (v1, v2) if v1 < v2 else (v2, v1)
564                    if key not in edge_dict:
565                        edge_dict[key] = len(edge_dict)
566                    local_edges.append(edge_dict[key])
567            cells2edges.append(local_edges)
568
569        # store on mesh for reuse
570        self.mesh.edges = np.array(list(edge_dict.keys()), dtype=int)
571        self.mesh.hypercells2edges = np.array(cells2edges, dtype=int)
572
573    def _finite_element(self, elnr):
574        """Return the P2 hypertriangle finite element template."""
575        return self.fe
576
577    def _element_dofs(self, elnr):
578        """
579        Local DOF numbering:
580        - first 5: vertex dofs (like P1)
581        - last 10: edge dofs (global index shifted by nv)
582        """
583        verts = self.mesh.elements()[elnr]
584        edge_ids = self.mesh.hypercells2edges[elnr]
585        return np.concatenate([verts, self.nv + edge_ids])
